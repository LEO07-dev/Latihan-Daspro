	kodeFak <-- char

deskripsi

	input kodeFak
	
	depend on (kodeFak)
		kodeFak = 'A' : output "fasilkom"
		kodeFak = 'B' : output "feb"
		kodeFak = 'C' : output "fib"
		kodeFak = 'D' : output "fkes"
		kodeFak = 'E' : output "fT""

		other : cout "Tidak ada"
